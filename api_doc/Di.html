set(id, funcOrValue) → {Di}

Source:
  • di.js, line 85
Set a value in the container
Examples

Set a raw value

di.set('color', '#ff0000')

Set a building function (value with be cached after first call)

di.set('database', function (di) {
  return new Database(di.get('database_url'));
})

Set a factory function (value will be factoryed each call)

di.set('token', di.factory(function () {
  return new Token();
}))

Set a building function that returns a promise

di.set('config', function () {
  return fsPromise.readFile('config.json');
})
